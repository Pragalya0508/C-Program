         NUMBER EVEN OR NOT 


#include <iostream>
using namespace std;

int main() {
  int n;

  cout << "Enter an integer: ";
  cin >> n;

  if ( n % 2 == 0)
    cout << n << " is even.";
  else
    cout << n << " is odd.";

  return 0;
}


OUTPUT:

Enter an integer: 23
23 is odd
