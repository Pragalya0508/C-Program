                            ADDING TWO NUMBERS



#include <iostream>  
using namespace std;  
int main() {  
    int num1, num2, sum;  
    cout << "Enter the first number: ";  
    cin >> num1;  
    cout << "Enter the second number: ";  
    cin >> num2;  
    sum = num1 + num2;  
    cout << "The sum of the two numbers is: " << sum;  
    return 0;  
}  


 OUTPUT:
  
       Enter the first number: 4
       Enter the second number: 7
       The sum of the two numbers is: 11
